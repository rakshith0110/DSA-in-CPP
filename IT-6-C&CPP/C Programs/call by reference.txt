
#include<stdio.h>		//Call by reference
void swapnum(int *n1,int *n2);
void main()
{
int num1,num2;
printf("\nEnter two numbers \n");
scanf("%d %d",&num1,&num2);	//num1=10, num2=40
swapnum(&num1,&num2);
printf("\nmain:After swapping num1=%d num2=%d", num1,num2);
}
void swapnum(int *n1,int *n2)
{
	int temp;
	temp=*n1;
	*n1=*n2;
	*n2=temp;
printf("\nswapnum: After swapping n1=%d n2=%d", *n1,*n2);
}


n1= &num1
*n1 --- Value stored in num1
             Value stored in addressed pointed by n1.
