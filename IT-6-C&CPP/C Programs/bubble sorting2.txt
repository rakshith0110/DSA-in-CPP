//Bubble sorting C++
#include<iostream>
int main()
{
    int arr[100];
    int i,j,temp, n;
    cout<<"Enter the size of array ";
    cin>>n;
    cout<<"Enter "<< n <<" numbers "<<endl;
    for(i=0;i<n;i++)
      cin>>&arr[i];

    for(i=0;i<n;i++)
    {
        for(j=0;j<n-1;j++)
        {
            if(arr[j]>arr[j+1])
            {
                temp=arr[j];
                arr[j]=arr[j+1];
                arr[j+1]=temp;
            }
        }
    }
    cout<<"Sorted elements are "<<endl;
    for(i=0;i<n;i++)
      cout<<arr[i]<<endl;
}
